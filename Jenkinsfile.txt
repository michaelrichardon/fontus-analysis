pipeline {
    agent any

    triggers {
        // Jeden Montag um 06:00 Uhr
        cron('0 6 * * 1')
    }

    environment {
        JIRA_BASE_URL           = 'https://jira.local.wmgruppe.de'
        JIRA_PROJECT            = 'FONTUS'
        JIRA_SSL_VERIFY         = 'false'
        JIRA_FEATURE_TEAM_FIELD = 'customfield_11101'
        JIRA_PAT                = credentials('jira-pat-fontus')
        JIRA_DB_PATH            = 'C:\\Jenkins\\fontus\\fontus_analysis.db'
        PYTHONIOENCODING        = 'utf-8'
    }

    stages {

        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Python-Umgebung') {
            steps {
                bat '''
                    python -m venv venv
                    venv\\Scripts\\pip install --upgrade pip --quiet
                    venv\\Scripts\\pip install -r requirements.txt --quiet
                '''
            }
        }

        stage('Analyse') {
            steps {
                bat '''
                    if not exist "C:\\Jenkins\\fontus" mkdir "C:\\Jenkins\\fontus"
                    set PYTHONIOENCODING=utf-8
                    venv\\Scripts\\python jira_analysis.py
                '''
            }
        }

        stage('Archivieren') {
            steps {
                archiveArtifacts(
                    artifacts: 'fontus_analysis_*.xlsx, fontus_transitions_*.json',
                    fingerprint: true,
                    allowEmptyArchive: false
                )
            }
        }

    }

    post {
        success {
            echo "Analyse erfolgreich abgeschlossen."
        }
        failure {
            echo "Analyse fehlgeschlagen – bitte Console Output pruefen."
        }
        always {
            bat 'if exist venv rmdir /s /q venv'
        }
    }
}
